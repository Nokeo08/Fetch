## Technical Notes

### Translation Function
```javascript
// Client-side
function t(key, params = {}) {
  const keys = key.split('.');
  let value = translations[currentLanguage];
  
  for (const k of keys) {
    value = value?.[k];
    if (!value) break;
  }
  
  // Fallback chain
  if (!value) {
    value = translations[defaultLanguage];
    for (const k of keys) {
      value = value?.[k];
      if (!value) break;
    }
  }
  
  if (!value) return key; // Return key as last resort
  
  // Parameter substitution
  return value.replace(/\{(\w+)\}/g, (match, param) => {
    return params[param] !== undefined ? params[param] : match;
  });
}

// Usage
t('items.added', { count: 5, name: 'Milk' });
// "Added 5 items including Milk"
```

### Loading Translations
```javascript
// Load based on detected language
async function loadTranslations(lang) {
  try {
    const response = await fetch(`/static/i18n/${lang}.json`);
    translations[lang] = await response.json();
    currentLanguage = lang;
  } catch (error) {
    console.error(`Failed to load translations for ${lang}`);
    currentLanguage = defaultLanguage;
  }
}
```

### Template Integration
```html
<!-- Server-side templating -->
<h1>{{ t("app.name") }}</h1>
<button>{{ t("items.add") }}</button>

<!-- Client-side -->
<h1 data-i18n="app.name"></h1>
<button data-i18n="items.add"></button>
```
